176  subroutine set_control_4_gravity(g_ctl, fl_prop)
177 !
178  use t_physical_property
179  use t_ctl_data_mhd_forces
180  use skip_comment_f
181 !
182  type(gravity_control), intent(in) :: g_ctl
183 !
184  type(fluid_property), intent(inout) :: fl_prop
185 !
186  integer (kind = kint) :: i
187  character(len=kchara) :: tmpchara
188 !
189 ! direction of gravity
190 !
191  fl_prop%i_grav = iflag_no_gravity
192  if( fl_prop%iflag_4_gravity &
193  & .or. fl_prop%iflag_4_composit_buo &
194  & .or. fl_prop%iflag_4_filter_gravity &
195  & .or. fl_prop%iflag_4_filter_comp_buo) then
196  if(g_ctl%FEM_gravity_model%iflag .gt. 0 &
197  & .and. cmp_no_case(g_ctl%FEM_gravity_model%charavalue,'node') &
198  & .and. fl_prop%iflag_scheme .ne. id_crank_nicolson_cmass) then
199  fl_prop%iflag_FEM_gravity = id_force_at_node
200  else
201  fl_prop%iflag_FEM_gravity = id_force_ele_int
202  end if
203 !
204 !
205  if (g_ctl%gravity%iflag .eq. 0) then
206  fl_prop%i_grav = iflag_self_r_g
207  else
208  tmpchara = g_ctl%gravity%charavalue
209 !
210  if (cmp_no_case(tmpchara, 'constant')) then
211  fl_prop%i_grav = iflag_const_g
212  else if(cmp_no_case(tmpchara, 'constant_radial')) then
213  fl_prop%i_grav = iflag_radial_g
214  else if(cmp_no_case(tmpchara, 'radial')) then
215  fl_prop%i_grav = iflag_self_r_g
216  end if
217  end if
218 !
219  if (fl_prop%i_grav .eq. iflag_const_g) then
220  if (g_ctl%gravity_vector%icou .eq. 0) then
221  e_message = 'Set gravity vector'
222  call calypso_mpi_abort(ierr_force, e_message)
223  else
224 !
225  do i = 1, g_ctl%gravity_vector%num
226  if(cmp_no_case(g_ctl%gravity_vector%c_tbl(i),'X') &
227  & ) fl_prop%grav(1) = - g_ctl%gravity_vector%vect(i)
228  if(cmp_no_case(g_ctl%gravity_vector%c_tbl(i),'Y') &
229  & ) fl_prop%grav(2) = - g_ctl%gravity_vector%vect(i)
230  if(cmp_no_case(g_ctl%gravity_vector%c_tbl(i),'Z') &
231  & ) fl_prop%grav(3) = - g_ctl%gravity_vector%vect(i)
232  end do
233  end if
234  end if
235  end if
236 !
237  if (iflag_debug .ge. iflag_routine_msg) then
238  write(*,*) 'i_grav ',fl_prop%i_grav
239  if(fl_prop%i_grav .eq. iflag_const_g) then
240  write(*,'(a, 1p3E25.15e3)') 'gravity ', fl_prop%grav(1:3)
241  end if
242  end if

244  end subroutine set_control_4_gravity
