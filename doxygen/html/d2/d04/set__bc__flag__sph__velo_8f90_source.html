143  j = sph_rj%idx_rj_degree_one( 0)
144  icb_uspec%Vt_BC(j) = r_icb*r_icb * bc_mag
145  end if
146 !

148  call set_fixed_vector_bc_by_file(velocity, sph_rj, bc_io, &
149  & sph_bc_u%icb_grp_name, sph_bc_u%iflag_icb, icb_uspec)
150  else if(ibc_type .eq. iflag_bc_evo_field) then
