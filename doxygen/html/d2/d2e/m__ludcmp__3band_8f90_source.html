8 ! SUBROUTINE ludcmp_3band(n, band_a, i_pivot, ierr, band_lu, d)
9 !
10 !c Given a matrix a(1:n,1:n), with physical dimension np by np,
11 !c this routine replaces it by the LU decomposition of a rowwise
12 !c permutation of itself. a and n are input.
13 !c a is output, arranged as in equation (2.3.14) above; index(1:n)
14 !c is an output vector that records the row permutation effected by
15 !c partial povoting; dis output as +/- 1 depending on whether
16 !c the number of row interchanges was even or odd, respectively.
17 !c This routine is used in combination with lubksb to solve
18 !c linear equations or ivert a matrix.
19 !
20 !
21 ! SUBROUTINE lubksb_3band(n, band_lu, i_pivot, x)
22 !
23 !c solve the set of n linear equations Ax=b. Here is a input,
24 !c not at the matrix A but rather as its LU decompsition,
25 !c determined by the routine ludcmp. i_pivot is input as the
26 !c permutation vector returned by ludcmp. b(1:n) is input
27 !c as the right-hand side vectror b, and returns with the
28 !c solution vector x.
29 !c a, n, np and i_pivot are not modified by this routine
30 !c and be left in place for successive calls with different
31 !c right hand sides b. This routine takes into account
32 !c the possibility that b will begin with many zero elements,
33 !c so it is efficient for use in matrix inversion.
34 !
35 !*
36 !* | a(2,1) a(1,2) ........ 0 0 |
37 !* | a(3,1) a(2,2) ........ . . |
38 !* | 0 a(3,2) ........ . . |
39 !* a(i,j) = | . 0 ........ 0 . |
40 !* | ...... a(3,k-1) a(2,k) a(1,k+1) .......... |
41 !* | . . ........ a(1,N-2) 0 |
42 !* | . . ........ a(2,N-2) a(1,N-1) |
43 !* | 0 0 ........ a(3,N-2) a(2,N-1) |
44 !
45 ! Original band matrix
46 ! band_a(i-j+iband+1,j) = a(i,j)
47 ! band_a(k,j) = a(k+j-iband-1,j)
48 ! 3-band matrix
49 ! band_a(i-j+2,j) = a(i,j)
50 ! band_a(k,j) = a(k+j-2,j)
51 ! 5-band matrix
52 ! band_lu(i-j+3,j) = a(i,j)
53 ! band_lu(k,j) = a(k+j-3,j)
54 ! 7-band matrix
55 ! band_lu(i-j+4,j) = a(i,j)
56 ! band_lu(k,j) = a(k+j-4,j)
57 ! 9-band matrix
58 ! band_lu(i-j+5,j) = a(i,j)
59 ! band_lu(k,j) = a(k+j-5,j)

74  SUBROUTINE ludcmp_3band(n, band_a, i_pivot, ierr, band_lu, d)
75 !
76  INTEGER (kind = kint), intent(in) :: n
77  REAL(kind = kreal), intent(in) :: band_a(3,n)
78 !
79  INTEGER(kind = kint), intent(inout) :: ierr
80  INTEGER(kind = kint), intent(inout) :: i_pivot(n)
81  REAL(kind = kreal), intent(inout) :: band_lu(5,n)
82 !
83  REAL(kind = kreal), intent(inout) :: d
84 !
85 !
86  INTEGER (kind = kint) :: i, j, k
87  INTEGER (kind = kint) :: kst, ked
88  REAL(kind = kreal) :: aamax,dum
89  integer(kind = kint), allocatable :: idx_org(:)
90  REAL(kind = kreal), allocatable :: vwk(:)
91 
92 !
93  allocate(vwk(n), idx_org(n))
94  vwk = 0.0d0
95 !
96  d = 1.0d0
97 !
98  band_lu = 0.0d0
99  band_lu(3,1) = band_a(2,1)
100  band_lu(4,1) = band_a(3,1)
101  do j = 2, n-1
102  band_lu(2,j) = band_a(1,j)
103  band_lu(3,j) = band_a(2,j)
104  band_lu(4,j) = band_a(3,j)
105  end do
106  band_lu(2,n) = band_a(1,n)
107  band_lu(3,n) = band_a(2,n)
108 !
109 !
110  do i = 1, n
111  aamax = abs(band_lu(3,i))
112  if(i .lt. n) aamax = max(aamax, abs(band_lu(itwo, i+ione)) )
113  if(i .gt. 1) aamax = max(aamax, abs(band_lu(ifour,i-ione)) )
114 !
115  if (aamax.eq.0.0d0) then
116  write(*,*) 'singular matrix in ludcmp'
117  ierr = 100
118  return
119  else
120  vwk(i) = 1./aamax
121  end if
122  idx_org(i) = i
123  end do
124 !
125  do j = 1, n
126 !
127  if(j .eq. 2) then
128  band_lu(3,j) = band_lu(3,j) - band_lu(4,j-1)*band_lu(2,j)
129  else if(j .gt. 2) then
130  band_lu(2,j) = band_lu(2,j) - band_lu(4,j-2)*band_lu(1,j)
131  band_lu(3,j) = band_lu(3,j) - band_lu(5,j-2)*band_lu(1,j) &
132  & - band_lu(4,j-1)*band_lu(2,j)
133  end if
134  i_pivot(j) = j
135 !
136  if (j .lt. n) then
137  aamax = vwk(j)*abs( band_lu(3,j) )
138 !
139  if(j.gt.1) band_lu(4,j) = band_lu(4,j) &
140  & - band_lu(5,j-1)*band_lu(2,j)
141  dum = vwk(j+1)*abs( band_lu(4,j) )
142 !
143  if (dum .ge. aamax .and. j.le.idx_org(j) ) then
144  k = idx_org(j+1)
145  idx_org(j+1) = idx_org(j)
146  idx_org(j) = k
147 !
148  i_pivot(j) = j+1
149  kst = max(ione, j-ione)
150  ked = min(j+itwo,n)
151  do k = kst, ked
152  dum = band_lu(j+4-k,k)
153  band_lu(j+4-k,k) = band_lu(j+3-k,k)
154  band_lu(j+3-k,k) = dum
155  end do
156  d = -d
157  vwk(j+1) = vwk(j)
158  end if
159  end if
160 !
161  if(band_lu(3,j).eq.0.0d0) band_lu(3,j) = tiny*tiny
162  dum = 1.0d0/band_lu(3,j)
163  if (j .lt. n) band_lu(4,j) = band_lu(4,j)*dum
164 !
165  end do
166 !
167  deallocate(vwk, idx_org)
168  ierr = 0
169 !
170  END subroutine ludcmp_3band
