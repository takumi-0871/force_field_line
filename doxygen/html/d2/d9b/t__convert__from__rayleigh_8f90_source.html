251  subroutine rescaling_from_rayleigh(l, m, nri_org, rayleigh_in)
252 !

256 !
257  integer(kind = kint), intent(in) :: l, m
258  integer(kind = kint), intent(in) :: nri_org

260 !
261  real(kind= kreal) :: pi
262 !
263 !

267  if(m .eq. 0) then

269  & = rayleigh_in(1:nri_org,1) * sqrt(two)

274  end if
275 !

308  subroutine radial_interpolation_rayleigh &
309  & (r_itp, nri_org, rayleigh_in, nri_tgt, rayleigh_tg)
310 !
