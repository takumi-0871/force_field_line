   uxuy = (ar) *   (               -s *up*ur + z   *ut*up)
   uzux = (aras) * ( r*x *ur*ut + y*z *up*ur + s*y *ut*up)
   uyuz = (aras) * (-r*y *ur*ut + x*z *up*ur + s*x *ut*up)
