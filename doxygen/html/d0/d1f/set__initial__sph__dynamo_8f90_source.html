299  subroutine set_initial_magne_sph(sph_rj, sph_bc_B, ipol, &
300  & r_ICB, r_CMB, nlayer_ICB, nlayer_CMB, &
301  & n_point, ntot_phys_rj, d_rj)
302 !
303  type(sph_boundary_type), intent(in) :: sph_bc_b
304  type(sph_rj_grid), intent(in) :: sph_rj
305  type(phys_address), intent(in) :: ipol
306  integer(kind = kint), intent(in) :: nlayer_icb, nlayer_cmb
307  real(kind = kreal), intent(in) :: r_icb, r_cmb
308  integer(kind = kint), intent(in) :: n_point, ntot_phys_rj
309  real (kind=kreal), intent(inout) :: d_rj(n_point,ntot_phys_rj)
310 !
311  real (kind = kreal) :: pi, rr
312  integer(kind = kint) :: is, it, k, js, jt
313 !
314 !
315  js = find_local_sph_address(sph_rj, 1,0)
316  jt = find_local_sph_address(sph_rj, 2,0)
317 !
318  pi = four * atan(one)
319 !
320 !$omp parallel do
321  do is = 1, n_point
322  d_rj(is,ipol%base%i_magne ) = zero
323  d_rj(is,ipol%base%i_magne+1) = zero
324  d_rj(is,ipol%base%i_magne+2) = zero
325  d_rj(is,ipol%base%i_current ) = zero
326  d_rj(is,ipol%base%i_current+1) = zero
327  d_rj(is,ipol%base%i_current+2) = zero
328  end do
329 !$omp end parallel do
330 !
331  if (sph_bc_b%iflag_icb .eq. iflag_sph_fill_center) then
332 !
333 ! Poloidal magnetic field
334  if (js .gt. 0) then
335  do k = nlayer_icb, nlayer_cmb
336  is = js + (k-1) * sph_rj%nidx_rj(2)
337  rr = sph_rj%radius_1d_rj_r(k)
338 !
339  d_rj(is,ipol%base%i_magne ) = (five / eight) &
340  & * (-three * rr**3 + four * r_cmb * rr**2 - r_icb**4/rr)
341  d_rj(is,ipol%base%i_magne+1) = (five / eight) &
342  & * (-dnine * rr**2 + eight * r_cmb * rr + r_icb**4/rr**2)
343  d_rj(is,ipol%base%i_current+2) = (five*three / two) * rr
344  end do
345  end if
346 !
347 ! Toroidal magnetic field
348  if (jt .gt. 0) then
349  do k = nlayer_icb, nlayer_cmb
350  it = jt + (k-1) * sph_rj%nidx_rj(2)
351  rr = sph_rj%radius_1d_rj_r(k)
352  d_rj(it,ipol%base%i_magne+2) &
353  & = (ten/three) * rr * sin(pi*(rr-r_icb))
354  d_rj(it,ipol%base%i_current ) &
355  & = d_rj(it,ipol%base%i_magne+2)
356  d_rj(it,ipol%base%i_current+1) &
357  & = (ten / three) * (sin(pi*(rr-r_icb)) &
358  & + pi * rr * cos(pi*(rr-r_icb)) )
359  end do
360  end if
361 !
362  else
363 !
364 ! Poloidal magnetic field
365  if (js .gt. 0) then
366  do k = 1, nlayer_cmb
367  is = js + (k-1) * sph_rj%nidx_rj(2)
368  rr = sph_rj%radius_1d_rj_r(k)
369  d_rj(is,ipol%base%i_magne ) = (five / two) * rr**2 &
370  & * (four*r_cmb - three*rr) / (r_cmb+three)
371  d_rj(is,ipol%base%i_magne+1) = (five / two) * rr &
372  & * (eight*r_cmb - dnine*rr) / (r_cmb+three)
373  d_rj(is,ipol%base%i_current+2) &
374  & = five*six * rr / (three +r_cmb)
375  end do
376  end if
377 !
378 ! Toroidal magnetic field
379  if (jt .gt. 0) then
380  do k = 1, nlayer_cmb
381  it = jt + (k-1) * sph_rj%nidx_rj(2)
382  rr = sph_rj%radius_1d_rj_r(k)
383 !
384  d_rj(it,ipol%base%i_magne+2) &
385  & = (ten / three) * rr * sin(pi*rr/r_cmb)
386  d_rj(it,ipol%base%i_current ) &
387  & = d_rj(it,ipol%base%i_magne+2)
388  d_rj(it,ipol%base%i_current+1) &
389  & = (ten / three) * (sin(pi*rr/r_cmb) &
390  & + (pi/r_cmb) * rr * cos(pi*rr/r_cmb) )
391  end do
392  end if
393 !
394  end if
395 !
396  end subroutine set_initial_magne_sph

401  subroutine reduce_initial_magne_sph &
402  & (ipol, nnod_rj, ntot_phys_rj, d_rj)
403 !

420  & = 1.0d-3 * d_rj(is,ipol%base%i_current )
421  d_rj(is,ipol%base%i_current+1) &
422  & = 1.0d-3 * d_rj(is,ipol%base%i_current+1)
423  d_rj(is,ipol%base%i_current+2) &
424  & = 1.0d-3 * d_rj(is,ipol%base%i_current+2)
425  end do
426 !$omp end parallel do
