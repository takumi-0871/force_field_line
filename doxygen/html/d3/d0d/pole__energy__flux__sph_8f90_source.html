223  & (numnod, internal_node, xx, nnod_rtp, nidx_rtp_r, coef, &
224  & ncomp_nod, i_temp, i_velo, i_buo_flux, d_nod)
225 !

228 !
229  integer(kind = kint), intent(in) :: nidx_rtp_r
230  integer(kind = kint), intent(in) :: nnod_rtp
231 !
232  integer(kind = kint), intent(in) :: ncomp_nod, i_buo_flux
233  integer(kind = kint), intent(in) :: i_temp, i_velo
234  real(kind = kreal), intent(in) :: coef
235  real(kind = kreal), intent(inout) :: d_nod(numnod, ncomp_nod)
236 !
237  integer(kind = kint) :: inod, kr
238 !
239 !
240  inod = nnod_rtp
241  if(inod .ge. internal_node) return
242 !
243 ! copy field for north pole
244  if(xx(inod+1,3) .gt. zero) then
245  do kr = 1, nidx_rtp_r
246  inod = inod + 1
247  d_nod(inod,i_buo_flux) &
248  & = coef*d_nod(inod,i_temp)*d_nod(inod,i_velo)*xx(inod,3)
249  end do
250  end if
251 !
252  if(inod .ge. internal_node) return
253 !
254 ! copy field for south pole
255  if(xx(inod+1,3) .lt. zero) then
256  do kr = 1, nidx_rtp_r
257  inod = inod + 1
258  d_nod(inod,i_buo_flux) &
259  & = - coef*d_nod(inod,i_temp)*d_nod(inod,i_velo)*xx(inod,3)
260  end do
261  end if
262 !
263  if(inod .ge. internal_node) return
264 !
265 ! copy field for center
266  inod = inod + 1
267  d_nod(inod,i_buo_flux) = zero
