252  subroutine add_reference_advect_sph &
253  & (sph_bc_s, property, ref_param_s, &
254  & nidx_rj, ar_1d_rj, g_sph_rj, is_h_advect, is_velo, &
255  & reftemp_rj, rj_fld)
256 !
257  integer(kind = kint), intent(in) :: nidx_rj(2)
258  integer(kind = kint), intent(in) :: is_h_advect, is_velo
259  real(kind = kreal), intent(in) :: g_sph_rj(nidx_rj(2),13)
260  real(kind = kreal), intent(in) :: ar_1d_rj(nidx_rj(1),3)
261  real(kind = kreal), intent(in) :: reftemp_rj(nidx_rj(1),0:2)
262  type(sph_boundary_type), intent(in) :: sph_bc_s
263  type(scalar_property), intent(in) :: property
264  type(reference_scalar_param), intent(in) :: ref_param_s
265 !
266  type(phys_data), intent(inout) :: rj_fld
267 !
268  integer(kind= kint) :: ist, ied, inod, j, k
269 !
270 !
271  if (ref_param_s%iflag_reference .ne. id_sphere_ref_temp) return
272 !
273  ist = (sph_bc_s%kr_in-1) * nidx_rj(2) + 1
274  ied = sph_bc_s%kr_out * nidx_rj(2)
275 !$omp parallel do private (inod,j,k)
276  do inod = ist, ied
277  j = mod((inod-1),nidx_rj(2)) + 1
278  k = 1 + (inod- j) / nidx_rj(2)
279 !
280  rj_fld%d_fld(inod,is_h_advect) = rj_fld%d_fld(inod,is_h_advect) &
281  & + property%coef_advect * g_sph_rj(j,3) * ar_1d_rj(k,2) &
282  & * reftemp_rj(k,1) * rj_fld%d_fld(inod,is_velo)
283  end do
284 !$omp end parallel do
285 !
286  end subroutine add_reference_advect_sph
