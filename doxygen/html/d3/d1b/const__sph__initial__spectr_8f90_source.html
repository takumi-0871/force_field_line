409  subroutine set_initial_magne_sph &
410  & (sph_bc_b, sph, ipol, rj_fld)
411 !
412  type(sph_boundary_type), intent(in) :: sph_bc_b
413  type(sph_grids), intent(in) :: sph
414  type(phys_address), intent(in) :: ipol
415  type(phys_data), intent(inout) :: rj_fld
416 !
417  real(kind = kreal), parameter :: b_adjust = 1.0e-3
418 !
419  real (kind = kreal) :: pi, rr
420  integer(kind = kint) :: is, it, is_icb, is_cmb
421  integer :: js, jt, k
422 !
423 !
424  if(ipol%base%i_magne .eq. izero) return
425  pi = four * atan(one)
426 !
427 !$omp parallel do
428  do is = 1, nnod_rj(sph)
429  rj_fld%d_fld(is,ipol%base%i_magne ) = zero
430  rj_fld%d_fld(is,ipol%base%i_magne+1) = zero
431  rj_fld%d_fld(is,ipol%base%i_magne+2) = zero
432  end do
433 !$omp end parallel do
434 !
435 !
436 ! Find local addrtess for (l,m) = (1,0)
437  js = find_local_sph_mode_address(sph, 1, 0)
438 !
439  if (js .gt. 0) then
440  do k = sph_bc_b%kr_in, sph_bc_b%kr_out
441  is = local_sph_data_address(sph, k, js)
442  rr = radius_1d_rj_r(sph, k)
443 ! Substitute poloidal mangetic field
444  rj_fld%d_fld(is,ipol%base%i_magne ) &
445  & = (5.0d0/8.0d0) * (-3.0d0 * rr**3 &
446  & + 4.0d0 * sph_bc_b%r_CMB(0) * rr**2 &
447  & - sph_bc_b%r_ICB(0)**4 / rr)
448  end do
449 !
450 ! Fill potential field if inner core exist
451  is_icb = local_sph_data_address(sph, int(sph_bc_b%kr_in), js)
452  do k = 1, sph_bc_b%kr_in-1
453  is = local_sph_data_address(sph, k, js)
454  rr = radius_1d_rj_r(sph, k) / sph_bc_b%r_ICB(0)
455 ! Substitute poloidal mangetic field
456  rj_fld%d_fld(is,ipol%base%i_magne ) &
457  & = rj_fld%d_fld(is_icb,ipol%base%i_magne ) * rr**(ione+1)
458  end do
459 !
460 ! Fill potential field if external of the core exist
461  is_cmb = local_sph_data_address(sph, int(sph_bc_b%kr_out), js)
462  do k = sph_bc_b%kr_out+1, nidx_rj(sph,1)
463  is = local_sph_data_address(sph, k, js)
464  rr = radius_1d_rj_r(sph, k) / sph_bc_b%r_CMB(0)
465 ! Substitute poloidal mangetic field
466  rj_fld%d_fld(is,ipol%base%i_magne ) &
467  & = rj_fld%d_fld(is_cmb,ipol%base%i_magne ) * rr**(-ione)
468  end do
469  end if
470 !
471 !
472 ! Find local addrtess for (l,m) = (2,0)
473  jt = find_local_sph_mode_address(sph, 1, -1)
474 !
475  if (jt .gt. 0) then
476  do k = 1, sph_bc_b%kr_out
477  it = local_sph_data_address(sph, k, jt)
478  rr = radius_1d_rj_r(sph, k)
479 ! Substitute totoidal mangetic field
480  rj_fld%d_fld(it,ipol%base%i_magne+2) &
481  & = (10.0d0/3.0d0) * rr * sin(pi*(rr - sph_bc_b%r_ICB(0)))
482  end do
483  end if
484 !
485 !
486 !$omp parallel do
487  do is = 1, nnod_rj(sph)
488  rj_fld%d_fld(is,ipol%base%i_magne ) &
489  & = b_adjust * rj_fld%d_fld(is,ipol%base%i_magne )
490  rj_fld%d_fld(is,ipol%base%i_magne+1) &
491  & = b_adjust * rj_fld%d_fld(is,ipol%base%i_magne+1)
492  rj_fld%d_fld(is,ipol%base%i_magne+2) &
493  & = b_adjust * rj_fld%d_fld(is,ipol%base%i_magne+2)
494  end do
495 !$omp end parallel do
496 !
497  end subroutine set_initial_magne_sph
