49 !

55  k_inter(k1,2) = 0
56  rcoef_inter(k1,1) = zero
57  rcoef_inter(k1,2) = one
58  else if(radius_1d_rj_r(k1) .gt. r_org(nri_org)) then
59  kr_outside = k1-1
60  exit
61  else
62  do k2 = kst, nri_org-1
63  if(radius_1d_rj_r(k1).ge.r_org(k2) &
64  & .and. radius_1d_rj_r(k1).le.r_org(k2+1)) then
65  k_inter(k1,1) = k2
66  k_inter(k1,2) = k2 + 1
67  rcoef_inter(k1,1) = (radius_1d_rj_r(k1) - r_org(k2)) &
68  & / (r_org(k2+1) - r_org(k2))
69  rcoef_inter(k1,2) = (r_org(k2+1) - radius_1d_rj_r(k1)) &
70  & / (r_org(k2+1) - r_org(k2))
71  kst = k2
72  exit
73  end if
74  end do
75  end if
76  end do
77  do k1 = kr_outside+1, nri_new
78  k_inter(k1,1) = nri_org+1
79  k_inter(k1,2) = nri_org+1
80  rcoef_inter(k1,1) = one
81  rcoef_inter(k1,2) = zero

132  end do
133  end do
