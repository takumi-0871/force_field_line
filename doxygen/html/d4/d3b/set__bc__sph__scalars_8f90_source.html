434  subroutine find_both_sides_of_boundaries(sph_rj, radial_rj_grp, &
435  & bc_nod, bc_surf, sph_bc, igrp_icb, igrp_cmb)
436 !
437  use t_bc_data_list
438 !
439  type(sph_rj_grid), intent(in) :: sph_rj
440  type(group_data), intent(in) :: radial_rj_grp
441  type(boundary_condition_list), intent(in) :: bc_nod, bc_surf
442 !
443  integer(kind = kint), intent(inout) :: igrp_icb, igrp_cmb
444  type(sph_boundary_type), intent(inout) :: sph_bc
445 !
446  integer(kind = kint) :: icou, i, j, inum, num, ibc_in, ibc_out
447  integer(kind = kint) :: igrp_bc(2), kr_bc(2)
448  character(len=kchara) :: r_bc_grp_name(2)
449 !
450 !
451  icou = 0
452  do i = 1, bc_nod%num_bc
453  if(icou .ge. 2) exit
454  do j = 1, radial_rj_grp%num_grp
455  if(bc_nod%bc_name(i) .eq. radial_rj_grp%grp_name(j)) then
456  num = radial_rj_grp%istack_grp(j) &
457  & - radial_rj_grp%istack_grp(j-1)
458  if(num .ne. 1) go to 10
459 !
460  icou = icou + 1
461  inum = radial_rj_grp%istack_grp(j)
462  kr_bc(icou) = radial_rj_grp%item_grp(inum)
463  r_bc_grp_name(icou) = bc_nod%bc_name(i)
464  igrp_bc(icou) = i
465  exit
466  end if
467  end do
468  end do
469 !
470  do i = 1, bc_surf%num_bc
