      subroutine legendre_f_trans_on_the_fly                          &
     &         (iflag_matmul, ncomp, nvector, nscalar,                &
     &          sph_rtm, sph_rlm, comm_rtm, comm_rlm, leg, idx_trns,  &
     &          n_WR, n_WS, WR, WS, WK_l_otf)

        Output: sp_rlm   (Order: poloidal,diff_poloidal,toroidal)
