   uxux = (as)^2 *(x*x *us*us - x*y *us*up
                 - x*y *us*up + y*y *up*up)
   uyuy = (as)^2 *(x*y *us*us + x*x *us*up
                 - y*y *us*up + x*x *up*up)
   uxuz = (as)^2 *(x*s *us*uz - y*s *up*uz)
   uyuy = (as)^2 *(y*y *us*us + x*y *us*up

   uyuz = (as)^2 *(y*s *us*uz + x*s *up*uz)
   uzuz = (as)^2 * s*s* uz*uz
