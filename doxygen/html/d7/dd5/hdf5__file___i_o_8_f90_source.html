88  subroutine copy_ele_connect_for_hdf5(ucd, ie_swap)
89 !
90  type(ucd_data), intent(in) :: ucd
91  integer(kind = kint_4b), intent(inout) &
92  & :: ie_swap(ucd%nnod_4_ele*ucd%nele)
93 !
94  integer(kind = kint) :: k1
95  integer(kind = kint_gl) :: iele, i
96 !
97 !
98 !$omp parallel do private(iele,k1,i)
99  do iele = 1, ucd%nele
100  do k1 = 1, ucd%nnod_4_ele
101  i = k1 + (iele-1) * ucd%nnod_4_ele
102  ie_swap(i) = int(ucd%ie(iele,k1)) - 1
103  end do
104  end do
105 !$omp end parallel do
106 !
107  end subroutine copy_ele_connect_for_hdf5

118  integer(kind = kint) :: inod

206  subroutine parallel_write_hdf5_mesh_file(file_prefix, ucd)
207 !
208  character(len = kchara), intent(in) :: file_prefix
209  type(ucd_data), intent(in) :: ucd
210 !
211 #ifdef HDF5_IO
212  character(len=kchara), parameter &
213  & :: node_dataset_name = "nodes"
214  character(len=kchara), parameter &
215  & :: elem_dataset_name = "elements"
216 !
217  integer(kind = kint) :: nnod
218  character(len = kchara) :: file_name
219  integer(hid_t) :: id_hdf5
220  integer(hid_t) :: plist_id ! Property list identifier
221  integer :: info
222  integer :: dataspace_dims
223  integer(hsize_t) :: node_dataspace_dim(2), elem_dataspace_dim(2)
224  integer(hid_t) :: node_dataspace_id, elem_dataspace_id
225  integer(hid_t) :: node_dataset_id, elem_dataset_id
226  integer(hsize_t) :: hyperslab_size(2), hyperslab_offset(2)
227  integer(hid_t) :: node_memory_dataspace, elem_memory_dataspace
228  integer(hid_t) :: node_file_dataspace, elem_file_dataspace
229  integer(hsize_t) :: buf_dims(2)
230  integer :: hdferr
231 !
232 !
233  nnod = int(ucd%istack_merged_intnod(my_rank+1) &
234  & - ucd%istack_merged_intnod(my_rank ))
235  allocate(fld_hdf5(9*nnod))
236  allocate(ie_hdf5(ucd%nnod_4_ele*ucd%nele))
237 !
238  file_name = set_merged_hdf_mesh_file_name(file_prefix)
239 !
240 ! Remove our own counts from the offset
241 !
242  info = mpi_info_null
243 !
244 ! Setup file access property list with parallel I/O access.
245 !
246  call h5pcreate_f(h5p_file_access_f, plist_id, hdferr)
247  call h5pset_fapl_mpio_f(plist_id, calypso_comm, info, hdferr)
248 !
249 ! Create new file collectively.
250 !
251  call h5fcreate_f(file_name, h5f_acc_trunc_f, id_hdf5, &
252  hdferr, access_prp = plist_id)
253  call h5pclose_f(plist_id, hdferr)
254 !
255 !
256 ! We first have to transpose the data so it fits correctly to the XDMF format
257 ! We also take this opportunity to use the "real" (C-like) indexing
258 !
259  call copy_node_position_for_hdf5(nnod, ucd, fld_hdf5(1))
260 !
261  call copy_ele_connect_for_hdf5(ucd, ie_hdf5)
262 !
263 ! Node data set dimensions are (number of nodes) x 3
264 !
265  dataspace_dims = 2
266  node_dataspace_dim(1) = n_vector
267  node_dataspace_dim(2) = ucd%istack_merged_intnod(nprocs)
268  call h5screate_simple_f(dataspace_dims, node_dataspace_dim, &
269  node_dataspace_id, hdferr)
270 !
271 ! Element data set dimensions are (number of elements) x (nodes per element)
272 !
273  dataspace_dims = 2
274  elem_dataspace_dim(1) = ucd%nnod_4_ele
275  elem_dataspace_dim(2) = ucd%istack_merged_ele(nprocs)
276  call h5screate_simple_f(dataspace_dims, elem_dataspace_dim, &
277  elem_dataspace_id, hdferr)
278 !
279 ! Create the datasets in the file
280 !
281  call h5dcreate_f(id_hdf5, node_dataset_name, h5t_native_double, &
282  node_dataspace_id, node_dataset_id, hdferr)
283  call h5dcreate_f(id_hdf5, elem_dataset_name, &
284  h5t_native_integer, elem_dataspace_id, elem_dataset_id, &
285  & hdferr)
286 !
287 ! We can close the created dataspaces now
288 !
289  call h5sclose_f(node_dataspace_id, hdferr)
290  call h5sclose_f(elem_dataspace_id, hdferr)
291 !
292 ! Create dataspace for memory hyperslab for nodes
293 !
294  hyperslab_size(1) = 3
295  hyperslab_size(2) = nnod
296  hyperslab_offset(1) = 0
297  hyperslab_offset(2) = ucd%istack_merged_intnod(my_rank)
298  call h5screate_simple_f(dataspace_dims, hyperslab_size, &
299  node_memory_dataspace, hdferr)
300 !
301 ! Create dataspace for file hyperslab for nodes
302 !
303  call h5dget_space_f(node_dataset_id, node_file_dataspace, &
304  & hdferr)
305  call h5sselect_hyperslab_f(node_file_dataspace, &
306  & h5s_select_set_f, hyperslab_offset, hyperslab_size, hdferr)
307 !
308 ! And similarly create file/memory hyperslabs for elements
309 !
310  hyperslab_size(1) = ucd%nnod_4_ele
311  hyperslab_size(2) = ucd%nele
312  hyperslab_offset(1) = 0
313  hyperslab_offset(2) = ucd%istack_merged_ele(my_rank)
314 
315  call h5screate_simple_f(dataspace_dims, hyperslab_size, &
316  elem_memory_dataspace, hdferr)
317 !
318  call h5dget_space_f(elem_dataset_id, elem_file_dataspace, &
319  & hdferr)
320  call h5sselect_hyperslab_f(elem_file_dataspace, &
321  h5s_select_set_f, hyperslab_offset, hyperslab_size, hdferr)
322 !
323 ! Create a property list for collective write
324 !
325  call h5pcreate_f(h5p_dataset_xfer_f, plist_id, hdferr)
326  call h5pset_dxpl_mpio_f(plist_id, h5fd_mpio_collective_f, hdferr)
327 !
328 ! Write the node data (finally!)
329 !
330  buf_dims(1) = 3
331  buf_dims(2) = nnod
332  call h5dwrite_f(node_dataset_id, h5t_native_double, &
333  fld_hdf5(1), buf_dims, hdferr, node_memory_dataspace, &
334  node_file_dataspace, plist_id)
335 !
336 ! And the element data
337 !
338  buf_dims(1) = ucd%nnod_4_ele
339  buf_dims(2) = ucd%nele
340  call h5dwrite_f(elem_dataset_id, h5t_native_integer, &
341  ie_hdf5(1), buf_dims, hdferr, elem_memory_dataspace, &
342  elem_file_dataspace, plist_id)
343 !
344 ! Close parallel access property list
345 !
346  call h5pclose_f(plist_id, hdferr)
347 !
348 ! Close the memory and file dataspaces
349 !
350  call h5sclose_f(node_memory_dataspace, hdferr)
351  call h5sclose_f(elem_memory_dataspace, hdferr)
352  call h5sclose_f(node_file_dataspace, hdferr)
353  call h5sclose_f(elem_file_dataspace, hdferr)
354 !
355 ! Finally close the datasets
356 !
357  call h5dclose_f(node_dataset_id, hdferr)
358  call h5dclose_f(elem_dataset_id, hdferr)
359 !
360 ! Close file
361  call h5fclose_f(id_hdf5, hdferr)
362 !
363 ! Deallocate transpose arrays
364  deallocate(ie_hdf5)
365  deallocate(fld_hdf5)
366 #endif
367 !
368  end subroutine parallel_write_hdf5_mesh_file

371 !
372  subroutine parallel_write_hdf5_field_file &
373  & (file_prefix, cur_step, ucd)
374 !
375  character(len = kchara), intent(in) :: file_prefix
376  integer(kind=kint), intent(in) :: cur_step
377  type(ucd_data), intent(in) :: ucd
378 !
379 #ifdef HDF5_IO
380  integer(kind = kint) :: istep, icou, nnod
381  character(len = kchara) :: file_name
382  integer(hid_t) :: id_hdf5
383  integer(hid_t) :: plist_id ! Property list identifier
384  integer(hid_t) :: field_dataspace_id
385  integer(hid_t) :: field_dataset_id
386  integer(hid_t) :: field_file_dataspace, field_memory_dataspace
387  integer :: info
388  integer(hsize_t) :: field_dataspace_dim(2)
389  integer :: dataspace_dims
390  integer :: hdferr
391  integer(hsize_t) :: buf_dims(2)
392  integer(hsize_t) :: hyperslab_size(2), hyperslab_offset(2)
393 !
394 ! Setup the filename
395 !
396  file_name = set_merged_hdf_field_file_name(file_prefix, cur_step)
397 !
398 ! Progress update
399 !
400  if (my_rank .eq. 0) write(*,*) 'merged HDF5 field data: ', &
401  & trim(file_name)
402 ! Remove our own counts from the offset
403 !
404  info = mpi_info_null
405 !
406 ! Setup file access property list with parallel I/O access.
407 !
408  call h5pcreate_f(h5p_file_access_f, plist_id, hdferr)
409  call h5pset_fapl_mpio_f(plist_id, calypso_comm, info, hdferr)
410 !
411 ! Create new file collectively.
412 !
413  call h5fcreate_f(file_name, h5f_acc_trunc_f, id_hdf5, &
414  hdferr, access_prp = plist_id)
415  call h5pclose_f(plist_id, hdferr)
416 !
417 ! Go through each of the fields
418 !
419  nnod = int(ucd%istack_merged_intnod(my_rank+1) &
420  & - ucd%istack_merged_intnod(my_rank ))
421  allocate(fld_hdf5(9*nnod))
422 !
423  icou = 0
424  do istep = 1, ucd%num_field, 1
425 !
426 ! Transpose the field data to fit XDMF standards
427 !
428  ncomp_hdf5 = n_scalar
429  if(ucd%num_comp(istep) .eq. n_scalar) then
430  ncomp_hdf5 = n_scalar
431  call copy_scalar_field_for_hdf5 &
432  & (nnod, icou, ucd, fld_hdf5(1))
433  else if(ucd%num_comp(istep) .eq. n_vector) then
434  ncomp_hdf5 = n_vector
435  call copy_vector_field_for_hdf5 &
436  & (nnod, icou, ucd, fld_hdf5(1))
437  else if(ucd%num_comp(istep) .eq. 6) then
438  ncomp_hdf5 = 9
439  call copy_sym_tensor_field_for_hdf5 &
440  & (nnod, icou, ucd, fld_hdf5(1))
441  end if
442  icou = icou + ucd%num_comp(istep)
443 !
444 ! Create a dataspace of the appropriate size
445 !
446  dataspace_dims = 2
447  field_dataspace_dim(1) = ncomp_hdf5
448  field_dataspace_dim(2) = ucd%istack_merged_intnod(nprocs)
449  call h5screate_simple_f(dataspace_dims, field_dataspace_dim, &
450  field_dataspace_id, hdferr)
451 
452 !
453 ! Create the dataset for the field
454 !
455  call h5dcreate_f(id_hdf5, trim(ucd%phys_name(istep)), &
456  & h5t_native_double, &
457  field_dataspace_id, field_dataset_id, hdferr)
458 !
459 ! Now we can close the dataspace
460 !
461  call h5sclose_f(field_dataspace_id, hdferr)
462 !
463 ! Create dataspace for memory hyperslab
464 !
465  hyperslab_size(1) = ncomp_hdf5
466  hyperslab_size(2) = nnod
467  hyperslab_offset(1) = 0
468  hyperslab_offset(2) = ucd%istack_merged_intnod(my_rank)
469  call h5screate_simple_f(dataspace_dims, hyperslab_size, &
470  field_memory_dataspace, hdferr)
471 !
472 ! Create dataspace for file hyperslab
473 !
474  call h5dget_space_f(field_dataset_id, field_file_dataspace, &
475  & hdferr)
476  call h5sselect_hyperslab_f(field_file_dataspace, &
477  h5s_select_set_f, hyperslab_offset, hyperslab_size, hdferr)
478 !
479 ! Create a property list for collective write
480 !
481  call h5pcreate_f(h5p_dataset_xfer_f, plist_id, hdferr)
482  call h5pset_dxpl_mpio_f(plist_id, h5fd_mpio_collective_f, &
483  & hdferr)
484 !
485 ! Write the field data
486 !
487  buf_dims(1) = ncomp_hdf5
488  buf_dims(2) = nnod
489  call h5dwrite_f(field_dataset_id, h5t_native_double, &
490  fld_hdf5(1), buf_dims, hdferr, &
491  field_memory_dataspace, field_file_dataspace, plist_id)
492 !
493 ! Close parallel access property list
494 !
495  call h5pclose_f(plist_id, hdferr)
496 !
497 ! Close the memory and file dataspaces
498 !
499  call h5sclose_f(field_memory_dataspace, hdferr)
500  call h5sclose_f(field_file_dataspace, hdferr)
501 !
502 ! Close the field dataset
503 !
504  call h5dclose_f(field_dataset_id, hdferr)
505  end do
506 !
507 ! Close file
508 !
509  call h5fclose_f(id_hdf5, hdferr)
510 !
511 ! Deallocate transpose arrays
512  deallocate(fld_hdf5)
513 #endif
514 !
515  end subroutine parallel_write_hdf5_field_file

568  subroutine parallel_write_xdmf_file(file_prefix, xdmf_dir_file, &

570 !

572  character(len = kchara), intent(in) :: xdmf_dir_file
