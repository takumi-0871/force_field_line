104  type(sph_rj_grid), intent(in) :: sph_rj
105  type(phys_data), intent(in) :: rj_fld
106 !
107  type(picked_spectrum_data), intent(in) :: gauss_coef
108  real(kind=kreal), intent(inout) &
109  & :: d_rj_out(gauss_coef%num_sph_mode_lc)
110 !
111  integer(kind = kint) :: inum, l, i
112  real(kind = kreal) :: r_icb, r_cmb, rcmb_to_re, ricb_to_rref
113  real(kind = kreal) :: r_4_gauss_coefs, a2r_4_gauss
114 !
115 !
116  r_4_gauss_coefs = gauss_coef%radius_gl(1)
117  r_icb = sph_rj%radius_1d_rj_r(sph_params%nlayer_ICB)
118  r_cmb = sph_rj%radius_1d_rj_r(sph_params%nlayer_CMB)
119  if(r_4_gauss_coefs .ge. r_cmb) then
120  a2r_4_gauss = one / (r_4_gauss_coefs**2)
121  rcmb_to_re = r_cmb / r_4_gauss_coefs
122 !$omp parallel do private(l,i)
123  do inum = 1, gauss_coef%num_sph_mode_lc
124  l = gauss_coef%idx_out(inum,1)
125  i = gauss_coef%idx_out(inum,4) &
126  & + (sph_params%nlayer_CMB-1) * sph_rj%nidx_rj(2)
127  d_rj_out(inum) = rj_fld%d_fld(i,ipol%base%i_magne) &
128  & * dble(l) * rcmb_to_re**l *a2r_4_gauss
129  end do
130 !$omp end parallel do
131 !
132  else if(r_4_gauss_coefs .le. r_icb) then
133  a2r_4_gauss = one / (r_icb**2)
134  ricb_to_rref = r_4_gauss_coefs / r_icb
135 !$omp parallel do private(l,i)
136  do inum = 1, gauss_coef%num_sph_mode_lc
137  l = gauss_coef%idx_out(inum,1)
138  i = gauss_coef%idx_out(inum,4) &
139  & + (sph_params%nlayer_ICB-1) * sph_rj%nidx_rj(2)
140  d_rj_out(inum) = - rj_fld%d_fld(i,ipol%base%i_magne) &
141  & * dble(l+1) * a2r_4_gauss * ricb_to_rref**(l-1)
142  end do
143 !$omp end parallel do
144  end if

146  end subroutine gauss_coefficients_4_write
