453  unsigned char inbuf[INBUFSIZ]; /* input buffer */
454  unsigned char outbuf[OUTBUFSIZ]; /* output buffer */
455 
456  FILE *fin, *fout; /* input and output file name */
457 
458  int count, flush, status;
459 
460  if ((fout = fopen(gz_file_name, "w")) == NULL) {
461  fprintf(stderr, "Can't open %s\n", txt_file_name);
462  exit(1);
463  }
464  if ((fin = fopen(txt_file_name, "r")) == NULL) {
465  fprintf(stderr, "Can't open %s\n", gz_file_name);
466  exit(1);
467  }
468 
469  /* memory manegement is controlled by zlib */
470  z.zalloc = Z_NULL;
471  z.zfree = Z_NULL;
472  z.opaque = Z_NULL;
473 
474  /* Initialization */
475  /* Second valuable: compression level between 0 and 9. 0 is non compress. */
476  /* Default is Z_DEFAULT_COMPRESSION (= 6) */
477  /*if (deflateInit(&z, Z_DEFAULT_COMPRESSION) != Z_OK) {
478  fprintf(stderr, "deflateInit: %s\n", (z.msg) ? z.msg : "???");
479  exit(1);
480  }*/
481  /* Construct gzip compatible header */
482  if (deflateInit2(&z, Z_DEFAULT_COMPRESSION, Z_DEFLATED, (Z_DEFAULT_COMPRESSION+32),
483  Z_DEFAULT_MEMLEVEL, Z_DEFAULT_STRATEGY) != Z_OK) {
484  fprintf(stderr, "deflateInit2: %s\n", (z.msg) ? z.msg : "???");
485  exit(1);
486  }
487 
488  z.avail_in = 0; /* nuber of byte in input buffer */
489  z.next_out = outbuf; /* set output pointer */
490  z.avail_out = OUTBUFSIZ; /* set size of output buffer */
491 
492  /* Normally, the second valuable of deflate() is set to Z_NO_FLUSH */
493  flush = Z_NO_FLUSH;
494 
495  while (1) {
496  if (z.avail_in == 0) { /* if input is finished */
497  z.next_in = inbuf; /* set input pointer to topof buffer */
498  z.avail_in = (uint) fread(inbuf, 1, INBUFSIZ, fin); /* read data */
499  /*printf("%s \n",inbuf);*/
500 
501  /* if input is end, set second valuable for deflate() to Z_FINISH */
502  if (z.avail_in < INBUFSIZ) flush = Z_FINISH;
503  }
504  status = deflate(&z, flush); /* compress */
505  if (status == Z_STREAM_END) break; /* stop */
506  if (status != Z_OK) { /* error */
507  fprintf(stderr, "deflate: %s\n", (z.msg) ? z.msg : "???");
508  exit(1);
509  }
510 
511  /* if output buffer is full, write data */
512  if (z.avail_out == 0) {
513  if (fwrite(outbuf, 1, OUTBUFSIZ, fout) != OUTBUFSIZ) {
514  fprintf(stderr, "Write error\n");
515  exit(1);
516  }
517  /* return output buffer to initial */
518  z.next_out = outbuf;
519  z.avail_out = OUTBUFSIZ;
520  }
521  }
522 
523  /* output rest data */

526  fprintf(stderr, "Write error\n");
527  exit(1);
528  }
529  }
530 
531  /* Finalize */
532  if (deflateEnd(&z) != Z_OK) {
533  fprintf(stderr, "deflateEnd: %s\n", (z.msg) ? z.msg : "???");
534  exit(1);
535  }
536 
537  fclose(fin);
538  fclose(fout);
539 
540  if(remove(txt_file_name) == 0){
541  printf(" %s is deleted.\n", txt_file_name);
542  } else {
543  printf(" Delete error \n");

545  };
