101  & - sin( theta(inod) )*sin( theta(inod) )) &
102  & - ttt * sin( theta(inod) ) * cos( theta(inod) )
103 !
104  tcyl(inod,3) &

107 !

112  & - ttp * sin( theta(inod) )
113 !

116  & - two*trt * cos( theta(inod) ) * sin( theta(inod) ) &
117  & + ttt * sin( theta(inod) ) * sin( theta(inod) )
118  end do
