311  & idx_global_rtp, idx_global_rtm, &
312  & idx_global_rlm, idx_global_rj)
313 !
314  integer(kind = kint), intent(in) :: nb
315  integer(kind = kint), intent(in) :: nnod_rtp, nnod_rtm
316  integer(kind = kint), intent(in) :: nnod_rlm, nnod_rj

337  end do

340  do nd = 1, nb

343  end do
344  end do
345 !

347  do nd = 1, nb
348  k = mod(nd-1,2) + 1
349  x_global_rj(nb*inod-nb+nd) = dble(nd*idx_global_rj(inod,k))
350  end do

359 !

361 !
362  integer(kind = kint), intent(in) :: nb, id_check

364 !
365 !

367  real(kind = kreal) :: diff
368 !
369 !
370  compare_transfer_sph_reals = 0
371  write(id_check,*) 'Wrong commnication in rtm => rtp with ', nb
372  do inod = 1, sph%sph_rtp%nnod_rtp
373  mphi = sph%sph_rtp%idx_global_rtp(inod,3)
374  if(abs(sph%sph_rtp%idx_gl_1d_rtp_p(mphi,2)) &
375  & .gt. sph%sph_params%l_truncation) cycle
376  if( idx_rtp_recieve(inod,1) .eq. 0 &
377  & .or. idx_rtp_recieve(inod,2) .eq. 0 &

382  end do

384  write(id_check,'(4i16,1p3E23.15)') inod, &

386  & x_rtp_recieve(nb*inod-nb+1), x_global_rtp(nb*inod-nb+1)
387  compare_transfer_sph_reals = 1
388  end if
389  end if
390  end do

392  write(id_check,*) 'Wrong commnication in rtp => rtm with ', nb

395  & .or. idx_rtm_recieve(inod,2) .eq. 0 &
396  & .or. idx_rtm_recieve(inod,3) .eq. 0) then

407  end do
