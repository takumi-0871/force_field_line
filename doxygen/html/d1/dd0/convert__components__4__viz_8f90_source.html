79 !
80  else if ( icomp_viz .eq. icomp_sphl_sym_tensor ) then
81  call cal_sph_tensor_smp(np_smp, nnod, istack_n_smp, &

83  & sph_r, cyl_s, a_r, a_s)
84 !
85  else if ( icomp_viz .eq. icomp_cyl_sym_tensor ) then
86  call cal_cyl_tensor_smp(np_smp, nnod, istack_n_smp, &

89 !
90  else if (icomp_viz.ge.icomp_xx &
91  & .and. icomp_viz.le.icomp_zz) then
92  call copy_nod_scalar_smp &
93  & (nnod, dat_xyz(1,icomp_viz), dat_viz(1,1))

107 !

109  call cal_cylinder_r_comp_smp(np_smp, nnod, istack_n_smp, &
110  & dat_xyz(1,1), dat_viz(1,1), xx(1,1), xx(1,2), cyl_s, a_s)
111 !
112 !
113  else if ( icomp_viz .eq. icomp_rr ) then
114  call cal_rr_tensor_smp(np_smp, nnod, istack_n_smp, &
115  & dat_xyz(1,1), dat_viz(1,1), xx(1,1), xx(1,2), xx(1,3), &
116  & sph_r, cyl_s, a_r)
117 !
118  else if ( icomp_viz .eq. icomp_rt ) then
119  call cal_rt_tensor_smp(np_smp, nnod, istack_n_smp, &
